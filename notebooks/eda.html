sorted_fv = np.sort(fv.values)
ranks     = np.linspace(0, 100, len(sorted_fv))

CANDIDATES = {
    85: {"color": "#f4a261", "ls": "--", "label": "P85 — ~15% spikes\n'mildly above average' bleeds in"},
    90: {"color": "#2a9d8f", "ls": "-",  "label": "P90 — ~10% spikes\nelbow is clearest here  ✓"},
    95: {"color": "#e76f51", "ls": ":",  "label": "P95 — ~5% spikes\nextreme only, recall drops"},
}

fig, ax = plt.subplots(figsize=(11, 5))

ax.plot(ranks, sorted_fv, lw=1.8, color="steelblue", zorder=2, label="σ_future (every tick, sorted)")

# shade the 'boring market' region
p85_val = fv.quantile(0.85)
ax.fill_between(ranks, sorted_fv,
                where=(sorted_fv <= p85_val),
                color="steelblue", alpha=0.08,
                label="Flat zone — baseline noise")

for p, style in CANDIDATES.items():
    val      = fv.quantile(p / 100)
    rate     = (fv > val).mean() * 100
    ax.axhline(val, color=style["color"], lw=1.4, ls=style["ls"], zorder=3)
    ax.axvline(p,   color=style["color"], lw=0.8, ls=style["ls"], zorder=3)
    ax.annotate(
        f"  p{p}  σ={val:.5f}\n  {rate:.0f}% labelled spike",
        xy=(p, val),
        xytext=(p - 30, val + (sorted_fv.max() - sorted_fv.min()) * (0.12 if p == 90 else 0.06)),
        fontsize=8.5, color=style["color"], fontweight="bold",
        arrowprops=dict(arrowstyle="->", color=style["color"], lw=0.9),
    )

ax.set_xlabel("Percentile rank  (0 = calmest tick, 100 = wildest)", fontsize=11)
ax.set_ylabel("σ_future   (60-second log-return std)", fontsize=11)
ax.set_title(
    "Where does the BTC volatility curve break?\n"
    "Flat = normal market.  Elbow = where spikes begin.  Choose your threshold at the bend.",
    fontsize=12,
)
ax.yaxis.set_major_formatter(mticker.FormatStrFormatter("%.5f"))
ax.legend(fontsize=9, loc="upper left")
ax.grid(alpha=0.25)
plt.tight_layout()
plt.savefig("../reports/eda_percentile_plot.png", dpi=150)
plt.show()
print("Saved → reports/eda_percentile_plot.png")
